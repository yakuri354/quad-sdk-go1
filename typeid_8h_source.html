24inline void erase_all(std::string &string, const std::string &search) {
25 for (size_t pos = 0;;) {
26 pos = string.find(search, pos);

28 string.erase(pos, search.length());

32PYBIND11_NOINLINE inline void clean_type_id(std::string &name) {

36 abi::__cxa_demangle(name.c_str(), nullptr, nullptr, &status), std::free };
