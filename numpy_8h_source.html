342template <typename T, ssize_t Dims>
343class unchecked_reference {
344protected:
345 static constexpr bool Dynamic = Dims < 0;
346 const unsigned char *data_;
347 // Storing the shape & strides in local variables (i.e. these arrays) allows the compiler to
348 // make large performance gains on big, nested loops, but requires compile-time dimensions
349 conditional_t<Dynamic, const ssize_t *, std::array<ssize_t, (size_t) Dims>>
350 shape_, strides_;
351 const ssize_t dims_;
352
353 friend class pybind11::array;
354 // Constructor for compile-time dimensions:
355 template <bool Dyn = Dynamic>
356 unchecked_reference(const void *data, const ssize_t *shape, const ssize_t *strides, enable_if_t<!Dyn, ssize_t>)
357 : data_{reinterpret_cast<const unsigned char *>(data)}, dims_{Dims} {
358 for (size_t i = 0; i < (size_t) dims_; i++) {
359 shape_[i] = shape[i];
360 strides_[i] = strides[i];
361 }
362 }
363 // Constructor for runtime dimensions:
364 template <bool Dyn = Dynamic>
365 unchecked_reference(const void *data, const ssize_t *shape, const ssize_t *strides, enable_if_t<Dyn, ssize_t> dims)
366 : data_{reinterpret_cast<const unsigned char *>(data)}, shape_{shape}, strides_{strides}, dims_{dims} {}
367
368public:
374 template <typename... Ix> const T &operator()(Ix... index) const {
375 static_assert(ssize_t{sizeof...(Ix)} == Dims || Dynamic,
376 "Invalid number of indices for unchecked array reference");
377 return *reinterpret_cast<const T *>(data_ + byte_offset_unsafe(strides_, ssize_t(index)...));
378 }
383 template <ssize_t D = Dims, typename = enable_if_t<D == 1 || Dynamic>>
384 const T &operator[](ssize_t index) const { return operator()(index); }
385
387 template <typename... Ix> const T *data(Ix... ix) const { return &operator()(ssize_t(ix)...); }
388
390 constexpr static ssize_t itemsize() { return sizeof(T); }
391
393 ssize_t shape(ssize_t dim) const { return shape_[(size_t) dim]; }
394
396 ssize_t ndim() const { return dims_; }
397
399 template <bool Dyn = Dynamic>
400 enable_if_t<!Dyn, ssize_t> size() const {
401 return std::accumulate(shape_.begin(), shape_.end(), (ssize_t) 1, std::multiplies<ssize_t>());
402 }
403 template <bool Dyn = Dynamic>
404 enable_if_t<Dyn, ssize_t> size() const {
405 return std::accumulate(shape_, shape_ + ndim(), (ssize_t) 1, std::multiplies<ssize_t>());
406 }
407
410 ssize_t nbytes() const {
411 return size() * itemsize();
412 }
413};

513 dtype strip_padding(ssize_t itemsize) {
514 // Recursively strip all void fields with empty names that are generated for
515 // padding fields (as of NumPy v1.11).
516 if (!has_fields())
517 return *this;
518
519 struct field_descr { PYBIND11_STR_TYPE name; object format; pybind11::int_ offset; };
520 std::vector<field_descr> field_descriptors;
521
522 for (auto field : attr("fields").attr("items")()) {
523 auto spec = field.cast<tuple>();
524 auto name = spec[0].cast<pybind11::str>();
525 auto format = spec[1].cast<tuple>()[0].cast<dtype>();
526 auto offset = spec[1].cast<tuple>()[1].cast<pybind11::int_>();
527 if (!len(name) && format.kind() == 'V')
528 continue;
529 field_descriptors.push_back({(PYBIND11_STR_TYPE) name, format.strip_padding(format.itemsize()), offset});
530 }
531
532 std::sort(field_descriptors.begin(), field_descriptors.end(),
533 [](const field_descr& a, const field_descr& b) {
534 return a.offset.cast<int>() < b.offset.cast<int>();
535 });
536
537 list names, formats, offsets;
538 for (auto& descr : field_descriptors) {
539 names.append(descr.name);
540 formats.append(descr.format);
541 offsets.append(descr.offset);
542 }
543 return dtype(names, formats, offsets, itemsize);
544 }

559 using ShapeContainer = detail::any_container<ssize_t>;
560 using StridesContainer = detail::any_container<ssize_t>;
561
562 // Constructs an array taking shape/strides from arbitrary container types
563 array(const pybind11::dtype &dt, ShapeContainer shape, StridesContainer strides,
564 const void *ptr = nullptr, handle base = handle()) {
565
566 if (strides->empty())
567 *strides = detail::c_strides(*shape, dt.itemsize());
568
569 auto ndim = shape->size();
570 if (ndim != strides->size())
571 pybind11_fail("NumPy: shape ndim doesn't match strides ndim");
572 auto descr = dt;
573
574 int flags = 0;
575 if (base && ptr) {
576 if (isinstance<array>(base))
577 /* Copy flags from base (except ownership bit) */
578 flags = reinterpret_borrow<array>(base).flags() & ~detail::npy_api::NPY_ARRAY_OWNDATA_;
579 else
580 /* Writable by default, easy to downgrade later on if needed */
581 flags = detail::npy_api::NPY_ARRAY_WRITEABLE_;
582 }
583
584 auto &api = detail::npy_api::get();
585 auto tmp = reinterpret_steal<object>(api.PyArray_NewFromDescr_(
586 api.PyArray_Type_, descr.release().ptr(), (int) ndim, shape->data(), strides->data(),
587 const_cast<void *>(ptr), flags, nullptr));
588 if (!tmp)
589 throw error_already_set();
590 if (ptr) {
591 if (base) {
592 api.PyArray_SetBaseObject_(tmp.ptr(), base.inc_ref().ptr());
593 } else {
594 tmp = reinterpret_steal<object>(api.PyArray_NewCopy_(tmp.ptr(), -1 /* any order */));
595 }
596 }
597 m_ptr = tmp.release().ptr();
598 }

1082inline PYBIND11_NOINLINE void register_structured_dtype(
1083 any_container<field_descriptor> fields,
1084 const std::type_info& tinfo, ssize_t itemsize,
1085 bool (*direct_converter)(PyObject *, void *&)) {
1086
1087 auto& numpy_internals = get_numpy_internals();
1088 if (numpy_internals.get_type_info(tinfo, false))
1089 pybind11_fail("NumPy: dtype is already registered");
1090
1091 // Use ordered fields because order matters as of NumPy 1.14:
1092 // https://docs.scipy.org/doc/numpy/release.html#multiple-field-indexing-assignment-of-structured-arrays
1093 std::vector<field_descriptor> ordered_fields(std::move(fields));
1094 std::sort(ordered_fields.begin(), ordered_fields.end(),
1095 [](const field_descriptor &a, const field_descriptor &b) { return a.offset < b.offset; });
1096
1097 list names, formats, offsets;
1098 for (auto& field : ordered_fields) {
1099 if (!field.descr)
1100 pybind11_fail(std::string("NumPy: unsupported field dtype: `") +
1101 field.name + "` @ " + tinfo.name());
1102 names.append(PYBIND11_STR_TYPE(field.name));
1103 formats.append(field.descr);
1104 offsets.append(pybind11::int_(field.offset));
1105 }
1106 auto dtype_ptr = pybind11::dtype(names, formats, offsets, itemsize).release().ptr();
1107
1108 // There is an existing bug in NumPy (as of v1.11): trailing bytes are
1109 // not encoded explicitly into the format string. This will supposedly
1110 // get fixed in v1.12; for further details, see these:
1111 // - https://github.com/numpy/numpy/issues/7797
1112 // - https://github.com/numpy/numpy/pull/7798
1113 // Because of this, we won't use numpy's logic to generate buffer format
1114 // strings and will just do it ourselves.
1115 ssize_t offset = 0;
1116 std::ostringstream oss;
1117 // mark the structure as unaligned with '^', because numpy and C++ don't
1118 // always agree about alignment (particularly for complex), and we're
1119 // explicitly listing all our padding. This depends on none of the fields
1120 // overriding the endianness. Putting the ^ in front of individual fields
1121 // isn't guaranteed to work due to https://github.com/numpy/numpy/issues/9049
1122 oss << "^T{";
1123 for (auto& field : ordered_fields) {
1124 if (field.offset > offset)
1125 oss << (field.offset - offset) << 'x';
1126 oss << field.format << ':' << field.name << ':';
1127 offset = field.offset + field.size;
1128 }
1129 if (itemsize > offset)
1130 oss << (itemsize - offset) << 'x';
1131 oss << '}';
1132 auto format_str = oss.str();
1133
1134 // Sanity check: verify that NumPy properly parses our buffer format string
1135 auto& api = npy_api::get();
1136 auto arr = array(buffer_info(nullptr, itemsize, format_str, 1));
1137 if (!api.PyArray_EquivTypes_(dtype_ptr, arr.dtype().ptr()))
1138 pybind11_fail("NumPy: invalid buffer descriptor!");
1139
1140 auto tindex = std::type_index(tinfo);
1141 numpy_internals.registered_dtypes[tindex] = { dtype_ptr, format_str };
1142 get_internals().direct_conversions[tindex].push_back(direct_converter);
1143}

1365enum class broadcast_trivial { non_trivial, c_trivial, f_trivial };
1366
1367// Populates the shape and number of dimensions for the set of buffers. Returns a broadcast_trivial
1368// enum value indicating whether the broadcast is "trivial"--that is, has each buffer being either a
1369// singleton or a full-size, C-contiguous (`c_trivial`) or Fortran-contiguous (`f_trivial`) storage
1370// buffer; returns `non_trivial` otherwise.
1371template <size_t N>
1372broadcast_trivial broadcast(const std::array<buffer_info, N> &buffers, ssize_t &ndim, std::vector<ssize_t> &shape) {
1373 ndim = std::accumulate(buffers.begin(), buffers.end(), ssize_t(0), [](ssize_t res, const buffer_info &buf) {
1374 return std::max(res, buf.ndim);
1375 });
1376
1377 shape.clear();
1378 shape.resize((size_t) ndim, 1);
1379
1380 // Figure out the output size, and make sure all input arrays conform (i.e. are either size 1 or
1381 // the full size).
1382 for (size_t i = 0; i < N; ++i) {
1383 auto res_iter = shape.rbegin();
1384 auto end = buffers[i].shape.rend();
1385 for (auto shape_iter = buffers[i].shape.rbegin(); shape_iter != end; ++shape_iter, ++res_iter) {
1386 const auto &dim_size_in = *shape_iter;
1387 auto &dim_size_out = *res_iter;
1388
1389 // Each input dimension can either be 1 or `n`, but `n` values must match across buffers
1390 if (dim_size_out == 1)
1391 dim_size_out = dim_size_in;
1392 else if (dim_size_in != 1 && dim_size_in != dim_size_out)
1393 pybind11_fail("pybind11::vectorize: incompatible size/dimension of inputs!");
1394 }
1395 }
1396
1397 bool trivial_broadcast_c = true;
1398 bool trivial_broadcast_f = true;
1399 for (size_t i = 0; i < N && (trivial_broadcast_c || trivial_broadcast_f); ++i) {
1400 if (buffers[i].size == 1)
1401 continue;
1402
1403 // Require the same number of dimensions:
1404 if (buffers[i].ndim != ndim)
1405 return broadcast_trivial::non_trivial;
1406
1407 // Require all dimensions be full-size:
1408 if (!std::equal(buffers[i].shape.cbegin(), buffers[i].shape.cend(), shape.cbegin()))
1409 return broadcast_trivial::non_trivial;
1410
1411 // Check for C contiguity (but only if previous inputs were also C contiguous)
1412 if (trivial_broadcast_c) {
1413 ssize_t expect_stride = buffers[i].itemsize;
1414 auto end = buffers[i].shape.crend();
1415 for (auto shape_iter = buffers[i].shape.crbegin(), stride_iter = buffers[i].strides.crbegin();
1416 trivial_broadcast_c && shape_iter != end; ++shape_iter, ++stride_iter) {
1417 if (expect_stride == *stride_iter)
1418 expect_stride *= *shape_iter;
1419 else
1420 trivial_broadcast_c = false;
1421 }
1422 }
1423
1424 // Check for Fortran contiguity (if previous inputs were also F contiguous)
1425 if (trivial_broadcast_f) {
1426 ssize_t expect_stride = buffers[i].itemsize;
1427 auto end = buffers[i].shape.cend();
1428 for (auto shape_iter = buffers[i].shape.cbegin(), stride_iter = buffers[i].strides.cbegin();
1429 trivial_broadcast_f && shape_iter != end; ++shape_iter, ++stride_iter) {
1430 if (expect_stride == *stride_iter)
1431 expect_stride *= *shape_iter;
1432 else
1433 trivial_broadcast_f = false;
1434 }
1435 }
1436 }
1437
1438 return
1439 trivial_broadcast_c ? broadcast_trivial::c_trivial :
1440 trivial_broadcast_f ? broadcast_trivial::f_trivial :
1441 broadcast_trivial::non_trivial;
1442}

1510struct vectorize_helper {
1511
1512// NVCC for some reason breaks if NVectorized is private
1513#ifdef __CUDACC__
1514public:
1515#else
1516private:
1517#endif
1518
1519 static constexpr size_t N = sizeof...(Args);
1520 static constexpr size_t NVectorized = constexpr_sum(vectorize_arg<Args>::vectorize...);
1521 static_assert(NVectorized >= 1,
1522 "pybind11::vectorize(...) requires a function with at least one vectorizable argument");
1523
1524public:
1525 template <typename T>
1526 explicit vectorize_helper(T &&f) : f(std::forward<T>(f)) { }
1527
1528 object operator()(typename vectorize_arg<Args>::type... args) {
1529 return run(args...,
1530 make_index_sequence<N>(),
1531 select_indices<vectorize_arg<Args>::vectorize...>(),
1532 make_index_sequence<NVectorized>());
1533 }
1534
1535private:
1536 remove_reference_t<Func> f;
1537
1538 // Internal compiler error in MSVC 19.16.27025.1 (Visual Studio 2017 15.9.4), when compiling with "/permissive-" flag
1539 // when arg_call_types is manually inlined.
1540 using arg_call_types = std::tuple<typename vectorize_arg<Args>::call_type...>;
1541 template <size_t Index> using param_n_t = typename std::tuple_element<Index, arg_call_types>::type;
1542
1543 using returned_array = vectorize_returned_array<Func, Return, Args...>;
1544
1545 // Runs a vectorized function given arguments tuple and three index sequences:
1546 // - Index is the full set of 0 ... (N-1) argument indices;
1547 // - VIndex is the subset of argument indices with vectorized parameters, letting us access
1548 // vectorized arguments (anything not in this sequence is passed through)
1549 // - BIndex is a incremental sequence (beginning at 0) of the same size as VIndex, so that
1550 // we can store vectorized buffer_infos in an array (argument VIndex has its buffer at
1551 // index BIndex in the array).
1552 template <size_t... Index, size_t... VIndex, size_t... BIndex> object run(
1553 typename vectorize_arg<Args>::type &...args,
1554 index_sequence<Index...> i_seq, index_sequence<VIndex...> vi_seq, index_sequence<BIndex...> bi_seq) {
1555
1556 // Pointers to values the function was called with; the vectorized ones set here will start
1557 // out as array_t<T> pointers, but they will be changed them to T pointers before we make
1558 // call the wrapped function. Non-vectorized pointers are left as-is.
1559 std::array<void *, N> params{{ &args... }};
1560
1561 // The array of `buffer_info`s of vectorized arguments:
1562 std::array<buffer_info, NVectorized> buffers{{ reinterpret_cast<array *>(params[VIndex])->request()... }};
1563
1564 /* Determine dimensions parameters of output array */
1565 ssize_t nd = 0;
1566 std::vector<ssize_t> shape(0);
1567 auto trivial = broadcast(buffers, nd, shape);
1568 auto ndim = (size_t) nd;
1569
1570 size_t size = std::accumulate(shape.begin(), shape.end(), (size_t) 1, std::multiplies<size_t>());
1571
1572 // If all arguments are 0-dimension arrays (i.e. single values) return a plain value (i.e.
1573 // not wrapped in an array).
1574 if (size == 1 && ndim == 0) {
1575 PYBIND11_EXPAND_SIDE_EFFECTS(params[VIndex] = buffers[BIndex].ptr);
1576 return cast(returned_array::call(f, *reinterpret_cast<param_n_t<Index> *>(params[Index])...));
1577 }
1578
1579 auto result = returned_array::create(trivial, shape);
1580
1581 if (size == 0) return std::move(result);
1582
1583 /* Call the function */
1584 auto mutable_data = returned_array::mutable_data(result);
1585 if (trivial == broadcast_trivial::non_trivial)
1586 apply_broadcast(buffers, params, mutable_data, size, shape, i_seq, vi_seq, bi_seq);
1587 else
1588 apply_trivial(buffers, params, mutable_data, size, i_seq, vi_seq, bi_seq);
1589
1590 return std::move(result);
1591 }
