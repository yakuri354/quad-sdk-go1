62 auto name_obj = reinterpret_steal<object>(PYBIND11_FROM_STRING(name));
63

65 issue no Python C API calls which could potentially invoke the
66 garbage collector (the GC will call type_traverse(), which will in
67 turn find the newly constructed type in an invalid state) */
68 auto heap_type = (PyHeapTypeObject *) PyType_Type.tp_alloc(&PyType_Type, 0);
69 if (!heap_type)

122extern "C" inline int pybind11_meta_setattro(PyObject* obj, PyObject* name, PyObject* value) {
123 // Use `_PyType_Lookup()` instead of `PyObject_GetAttr()` in order to get the raw
124 // descriptor (`property`) instead of calling `tp_descr_get` (`property.__get__()`).
125 PyObject *descr = _PyType_Lookup((PyTypeObject *) obj, name);
126
127 // The following assignment combinations are possible:
128 // 1. `Type.static_prop = value` --> descr_set: `Type.static_prop.__set__(value)`
129 // 2. `Type.static_prop = other_static_prop` --> setattro: replace existing `static_prop`
130 // 3. `Type.regular_attribute = value` --> setattro: regular attribute assignment
131 const auto static_prop = (PyObject *) get_internals().static_property_type;
132 const auto call_descr_set = descr && PyObject_IsInstance(descr, static_prop)
133 && !PyObject_IsInstance(value, static_prop);
134 if (call_descr_set) {
135 // Call `static_property.__set__()` instead of replacing the `static_property`.
136#if !defined(PYPY_VERSION)
137 return Py_TYPE(descr)->tp_descr_set(descr, obj, value);
138#else
139 if (PyObject *result = PyObject_CallMethod(descr, "__set__", "OO", obj, value)) {
140 Py_DECREF(result);
141 return 0;
142 } else {
143 return -1;
144 }
145#endif
146 } else {
147 // Replace existing attribute.
148 return PyType_Type.tp_setattro(obj, name, value);
149 }
150}

172extern "C" inline PyObject *pybind11_meta_call(PyObject *type, PyObject *args, PyObject *kwargs) {
173
174 // use the default metaclass call to create/initialize the object
175 PyObject *self = PyType_Type.tp_call(type, args, kwargs);
176 if (self == nullptr) {
177 return nullptr;
178 }
179
180 // This must be a pybind11 instance
181 auto instance = reinterpret_cast<detail::instance *>(self);
182
183 // Ensure that the base __init__ function(s) were called
184 for (const auto &vh : values_and_holders(instance)) {
185 if (!vh.holder_constructed()) {
186 PyErr_Format(PyExc_TypeError, "%.200s.__init__() must be called when overriding __init__",
187 get_fully_qualified_tp_name(vh.type->type).c_str());
188 Py_DECREF(self);
189 return nullptr;
190 }
191 }
192
193 return self;
194}

240inline void traverse_offset_bases(void *valueptr, const detail::type_info *tinfo, instance *self,
241 bool (*f)(void * /*parentptr*/, instance * /*self*/)) {
242 for (handle h : reinterpret_borrow<tuple>(tinfo->type->tp_bases)) {
243 if (auto parent_tinfo = get_type_info((PyTypeObject *) h.ptr())) {
244 for (auto &c : parent_tinfo->implicit_casts) {
245 if (c.first == tinfo->cpptype) {
246 auto *parentptr = c.second(valueptr);
247 if (parentptr != valueptr)
248 f(parentptr, self);
249 traverse_offset_bases(parentptr, parent_tinfo, self, f);
250 break;
251 }
252 }
253 }
254 }
255}

257inline bool register_instance_impl(void *ptr, instance *self) {
258 get_internals().registered_instances.emplace(ptr, self);
259 return true; // unused, but gives the same signature as the deregister func
260}
261inline bool deregister_instance_impl(void *ptr, instance *self) {
262 auto &registered_instances = get_internals().registered_instances;
263 auto range = registered_instances.equal_range(ptr);
264 for (auto it = range.first; it != range.second; ++it) {
265 if (self == it->second) {
266 registered_instances.erase(it);
267 return true;
268 }
269 }
270 return false;
271}

273inline void register_instance(instance *self, void *valptr, const type_info *tinfo) {
274 register_instance_impl(valptr, self);
275 if (!tinfo->simple_ancestors)
276 traverse_offset_bases(valptr, tinfo, self, register_instance_impl);
277}
278
279inline bool deregister_instance(instance *self, void *valptr, const type_info *tinfo) {
280 bool ret = deregister_instance_impl(valptr, self);
281 if (!tinfo->simple_ancestors)
282 traverse_offset_bases(valptr, tinfo, self, deregister_instance_impl);
283 return ret;
284}

289inline PyObject *make_new_instance(PyTypeObject *type) {
290#if defined(PYPY_VERSION)
291 // PyPy gets tp_basicsize wrong (issue 2482) under multiple inheritance when the first inherited
292 // object is a a plain Python type (i.e. not derived from an extension type). Fix it.
293 ssize_t instance_size = static_cast<ssize_t>(sizeof(instance));
294 if (type->tp_basicsize < instance_size) {
295 type->tp_basicsize = instance_size;
296 }
297#endif
298 PyObject *self = type->tp_alloc(type, 0);
299 auto inst = reinterpret_cast<instance *>(self);
300 // Allocate the value/holder internals:
301 inst->allocate_layout();
302
303 inst->owned = true;
304
305 return self;
306}

332inline void clear_patients(PyObject *self) {
333 auto instance = reinterpret_cast<detail::instance *>(self);
334 auto &internals = get_internals();
335 auto pos = internals.patients.find(self);
336 assert(pos != internals.patients.end());
337 // Clearing the patients can cause more Python code to run, which
338 // can invalidate the iterator. Extract the vector of patients
339 // from the unordered_map first.
340 auto patients = std::move(pos->second);
341 internals.patients.erase(pos);
342 instance->has_patients = false;
343 for (PyObject *&patient : patients)
344 Py_CLEAR(patient);
345}

349inline void clear_instance(PyObject *self) {
350 auto instance = reinterpret_cast<detail::instance *>(self);
351
352 // Deallocate any values/holders, if present:
353 for (auto &v_h : values_and_holders(instance)) {
354 if (v_h) {
355
356 // We have to deregister before we call dealloc because, for virtual MI types, we still
357 // need to be able to get the parent pointers.
358 if (v_h.instance_registered() && !deregister_instance(instance, v_h.value_ptr(), v_h.type))
359 pybind11_fail("pybind11_object_dealloc(): Tried to deallocate unregistered instance!");
360
361 if (instance->owned || v_h.holder_constructed())
362 v_h.type->dealloc(v_h);
363 }
364 }
365 // Deallocate the value/holder layout internals:
366 instance->deallocate_layout();
367
368 if (instance->weakrefs)
369 PyObject_ClearWeakRefs(self);
370
371 PyObject **dict_ptr = _PyObject_GetDictPtr(self);
372 if (dict_ptr)
373 Py_CLEAR(*dict_ptr);
374
375 if (instance->has_patients)
376 clear_patients(self);
377}

381extern "C" inline void pybind11_object_dealloc(PyObject *self) {
382 clear_instance(self);
383
384 auto type = Py_TYPE(self);
385 type->tp_free(self);
386
387#if PY_VERSION_HEX < 0x03080000
388 // `type->tp_dealloc != pybind11_object_dealloc` means that we're being called
389 // as part of a derived type's dealloc, in which case we're not allowed to decref
390 // the type here. For cross-module compatibility, we shouldn't compare directly
391 // with `pybind11_object_dealloc`, but with the common one stashed in internals.
392 auto pybind11_object_type = (PyTypeObject *) get_internals().instance_base;
393 if (type->tp_dealloc == pybind11_object_type->tp_dealloc)
394 Py_DECREF(type);
395#else
396 // This was not needed before Python 3.8 (Python issue 35810)
397 // https://github.com/pybind/pybind11/issues/1946
398 Py_DECREF(type);
399#endif
400}

499extern "C" inline int pybind11_getbuffer(PyObject *obj, Py_buffer *view, int flags) {
500 // Look for a `get_buffer` implementation in this type's info or any bases (following MRO).
501 type_info *tinfo = nullptr;
502 for (auto type : reinterpret_borrow<tuple>(Py_TYPE(obj)->tp_mro)) {
503 tinfo = get_type_info((PyTypeObject *) type.ptr());
504 if (tinfo && tinfo->get_buffer)
505 break;
506 }
507 if (view == nullptr || !tinfo || !tinfo->get_buffer) {
508 if (view)
509 view->obj = nullptr;
510 PyErr_SetString(PyExc_BufferError, "pybind11_getbuffer(): Internal error");
511 return -1;
512 }
513 std::memset(view, 0, sizeof(Py_buffer));
514 buffer_info *info = tinfo->get_buffer(obj, tinfo->get_buffer_data);
515 view->obj = obj;
516 view->ndim = 1;
517 view->internal = info;
518 view->buf = info->ptr;
519 view->itemsize = info->itemsize;
520 view->len = view->itemsize;
521 for (auto s : info->shape)
522 view->len *= s;
523 view->readonly = info->readonly;
524 if ((flags & PyBUF_WRITABLE) == PyBUF_WRITABLE && info->readonly) {
525 if (view)
526 view->obj = nullptr;
527 PyErr_SetString(PyExc_BufferError, "Writable buffer requested for readonly storage");
528 return -1;
529 }
530 if ((flags & PyBUF_FORMAT) == PyBUF_FORMAT)
531 view->format = const_cast<char *>(info->format.c_str());
532 if ((flags & PyBUF_STRIDES) == PyBUF_STRIDES) {
533 view->ndim = (int) info->ndim;
534 view->strides = &info->strides[0];
535 view->shape = &info->shape[0];
536 }
537 Py_INCREF(view->obj);
538 return 0;
539}

559inline PyObject* make_new_python_type(const type_record &rec) {
560 auto name = reinterpret_steal<object>(PYBIND11_FROM_STRING(rec.name));
561
562 auto qualname = name;
563 if (rec.scope && !PyModule_Check(rec.scope.ptr()) && hasattr(rec.scope, "__qualname__")) {
564#if PY_MAJOR_VERSION >= 3
565 qualname = reinterpret_steal<object>(
566 PyUnicode_FromFormat("%U.%U", rec.scope.attr("__qualname__").ptr(), name.ptr()));
567#else
568 qualname = str(rec.scope.attr("__qualname__").cast<std::string>() + "." + rec.name);
569#endif
570 }
571
572 object module_;
573 if (rec.scope) {
574 if (hasattr(rec.scope, "__module__"))
575 module_ = rec.scope.attr("__module__");
576 else if (hasattr(rec.scope, "__name__"))
577 module_ = rec.scope.attr("__name__");
578 }
579
580 auto full_name = c_str(
581#if !defined(PYPY_VERSION)
582 module_ ? str(module_).cast<std::string>() + "." + rec.name :
583#endif
584 rec.name);
585
586 char *tp_doc = nullptr;
587 if (rec.doc && options::show_user_defined_docstrings()) {
588 /* Allocate memory for docstring (using PyObject_MALLOC, since
589 Python will free this later on) */
590 size_t size = strlen(rec.doc) + 1;
591 tp_doc = (char *) PyObject_MALLOC(size);
592 memcpy((void *) tp_doc, rec.doc, size);
593 }
594
595 auto &internals = get_internals();
596 auto bases = tuple(rec.bases);
597 auto base = (bases.empty()) ? internals.instance_base
598 : bases[0].ptr();
599
600 /* Danger zone: from now (and until PyType_Ready), make sure to
601 issue no Python C API calls which could potentially invoke the
602 garbage collector (the GC will call type_traverse(), which will in
603 turn find the newly constructed type in an invalid state) */
604 auto metaclass = rec.metaclass.ptr() ? (PyTypeObject *) rec.metaclass.ptr()
605 : internals.default_metaclass;
606
607 auto heap_type = (PyHeapTypeObject *) metaclass->tp_alloc(metaclass, 0);
608 if (!heap_type)
609 pybind11_fail(std::string(rec.name) + ": Unable to create type object!");
610
611 heap_type->ht_name = name.release().ptr();
612#ifdef PYBIND11_BUILTIN_QUALNAME
613 heap_type->ht_qualname = qualname.inc_ref().ptr();
614#endif
615
616 auto type = &heap_type->ht_type;
617 type->tp_name = full_name;
618 type->tp_doc = tp_doc;
619 type->tp_base = type_incref((PyTypeObject *)base);
620 type->tp_basicsize = static_cast<ssize_t>(sizeof(instance));
621 if (!bases.empty())
622 type->tp_bases = bases.release().ptr();
623
624 /* Don't inherit base __init__ */
625 type->tp_init = pybind11_object_init;
626
627 /* Supported protocols */
628 type->tp_as_number = &heap_type->as_number;
629 type->tp_as_sequence = &heap_type->as_sequence;
630 type->tp_as_mapping = &heap_type->as_mapping;
631#if PY_VERSION_HEX >= 0x03050000
632 type->tp_as_async = &heap_type->as_async;
633#endif
634
635 /* Flags */
636 type->tp_flags |= Py_TPFLAGS_DEFAULT | Py_TPFLAGS_HEAPTYPE;
637#if PY_MAJOR_VERSION < 3
638 type->tp_flags |= Py_TPFLAGS_CHECKTYPES;
639#endif
640 if (!rec.is_final)
641 type->tp_flags |= Py_TPFLAGS_BASETYPE;
642
643 if (rec.dynamic_attr)
644 enable_dynamic_attributes(heap_type);
645
646 if (rec.buffer_protocol)
647 enable_buffer_protocol(heap_type);
648
649 if (PyType_Ready(type) < 0)
