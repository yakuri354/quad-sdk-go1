154inline void finalize_interpreter() {
155 handle builtins(PyEval_GetBuiltins());
156 const char *id = PYBIND11_INTERNALS_ID;
157
158 // Get the internals pointer (without creating it if it doesn't exist). It's possible for the
159 // internals to be created during Py_Finalize() (e.g. if a py::capsule calls `get_internals()`
160 // during destruction), so we get the pointer-pointer here and check it after Py_Finalize().
161 detail::internals **internals_ptr_ptr = detail::get_internals_pp();
162 // It could also be stashed in builtins, so look there too:
163 if (builtins.contains(id) && isinstance<capsule>(builtins[id]))
164 internals_ptr_ptr = capsule(builtins[id]);
165
166 Py_Finalize();
167
168 if (internals_ptr_ptr) {
169 delete *internals_ptr_ptr;
170 *internals_ptr_ptr = nullptr;
171 }
172}
